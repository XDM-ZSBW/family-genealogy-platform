# scripts/rename_s_letter_to_wiki.py
import shutil
from pathlib import Path
import re

def update_wikilinks(file_path):
    """Update all wikilinks in a markdown file"""
    content = file_path.read_text(encoding='utf-8')
    # Replace s_letter with wiki in all links
    updated = content.replace('s_letter/', 'wiki/')
    updated = updated.replace('[[s_letter/', '[[wiki/')
    file_path.write_text(updated, encoding='utf-8')

def main():
    base = Path("Gladys")

    # Rename directory
    old_path = base / "s_letter"
    new_path = base / "wiki"
    shutil.move(str(old_path), str(new_path))

    # Update all markdown files
    for md_file in base.rglob("*.md"):
        update_wikilinks(md_file)
